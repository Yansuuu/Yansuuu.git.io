制作一个简单的todo应用

HTML部分

<body ng-app="App">
    <div class="wrap" ng-controller="todoController"> 
        <form ng-submit="addItme()">
            <lable>添加待办事项</lable>
            <input type="text" ng-model="todo">
        </form>
        <dl>
            <dt>待办事宜</dt>
            <dl ng-repeat="todo in todos">
                <input type="checkbox" ng-checked="todo.checked" ng-click="done($index, $event)">{{todo.text}}
                <a href="javascript:;" ng-click="deleteItem($index, todos)">删除</a>
            </dl>
            <dt>已办理事情{{dones.length}}</dt>
            <dl ng-repeat=" done in dones ">
                <input type="checkbox" ng-checked="done.checked" ng-click="undone($index, $event)">{{done.text}}
                <a href="javascript:;" ng-click="deleteItem($index, dones)">删除</a>
            </dl>
        </dl>
    </div>

    <!--引入框架-->
    <script src="../libs/angular.min.js"></script>
</body>

javascript部分

angular.module( "App", [] )
    .controller( "todoController", [ "$scope", function ($scope) {

        // 都是在操作这两个数组
        // 并将两个数组的变化反映在页面上
        $scope.todos = [];
        $scope.dones = [];

        $scope.addItme = function () {
            // 当表单提交事件出发的时候，todos数组中添加一个对象
            // 并清空输入框
            $scope.todos.push( {text: $scope.todo, checked: false} );
            $scope.todo = "";
        };

        $scope.done = function ( index, ev ) {
            // console.log(index);

            // console.log(ev);
            // 将对应项从todos数组中删除，并拼接到dones数组中
            var temp = $scope.todos.splice( index, 1 );
            // 将得到的数组中的项的checked属性改为相反
            temp[0].checked = !temp[0].checked;

            // concat方法返回一个新数组，并不修改原数组
            $scope.dones = $scope.dones.concat( temp );
            // 阻止浏览器默认行为
            ev.preventDefault();
        };

        $scope.undone = function ( index, e ) {
            // 上面地相反操作
            var temp = $scope.dones.splice( index, 1 );
            // 将得到的数组中的项的checked属性改为相反
            temp[0].checked = !temp[0].checked;

            // concat方法返回一个新数组，并不修改原数组
            // push进去的是temp中的那个对象，tamp本身是个数组
            $scope.todos.push( temp[0] );
            // 阻止浏览器默认行为
            ev.preventDefault();
        };


        // 从当前循环的数组中删除当前项
        $scope.deleteItem = function ( index, things ) {
            things.splice( index, 1 );
        }

    } ] );

js部分

    angular.module( "myApp", [] )
        .controller( "tabsController", [ "$scope", function ($scope) {

            // 点击标签的时候显示对应tab
            $scope.switchTab = function (type) {
                $scope.type = type;

                window.event.preventDefault();
            }

            // 默认是国内 这个案例操作的数据就是个type
            $scope.type = "local";
        } ] );
