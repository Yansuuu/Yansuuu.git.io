js部分

// 创建模块对象
var App = angular.module( "App", [ "ngRoute" ] );

// 创建控制器对象
App.controller( "navController", [ "$scope", "$http", function ( $scope, $http ) { 


    // 还是对dom的操作 实现点击变色 其他导航恢复原色
    $scope.change = function ( e ) {
        var target = e.target;
        // target.style.background = "#C63";
        var siblings = target.parentNode.childNodes;

        var newSiblings = [];
        // 过滤出元素节点
        [].forEach.call( siblings, function ( val ) { 

            if ( val.nodeType === 1 ) {
                newSiblings.push( val );
            }

        } )

        newSiblings.forEach( function ( val ) { 

            val.classList.remove( 'active' );

        } )


        target.classList.add( 'active' );
        // console.log(siblings);

    }

    // 发送请求获取数据
    $http({

        url: './api/nav.php',

    }).success( function ( data ) { 

        $scope.navs = data;

    } );

} ] );

App.config([ "$routeProvider", function ( $routeProvider ) { 

    // 这里监听hash的变化
    $routeProvider.when( '/:id', {
        templateUrl: './views/songlist.html',
        controller: 'songController'
    } ).otherwise({
        redirectTo: '/1'
    })

} ]);

App.controller( 'songController', [ "$scope", "$http", "$routeParams", function ( $scope, $http, $routeParams ) { 

    var pid = $routeParams.id || 1;

    $http( {

        url: './api/song.php',
        method: 'get',
        params: {
            pid: pid
        }

    } ).success( function ( data ) { 

        $scope.songs = data;

    } )

} ] );
