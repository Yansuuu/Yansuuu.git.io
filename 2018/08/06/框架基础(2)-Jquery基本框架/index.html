jQuery基本框架

jQuery和$的实质是一个返回实例的工厂函数

通过一个沙箱将整个模块包围，只暴露的工厂函数

//1.70版本的
(functon (w) {
    //工厂函数
 function jQuery(selector) {
    //工厂函数返回一个实例对象
    return new jQuery.fn.init(selector);
}
//简化原型写法
jQuery.fn = jQuery.prototype = {
    constrctor: jQuery
}
//真正的构造函数在工厂函数的原型上
var init = jQuery.fn.init = function (selecor) {

}
//将构造函数的原型替换为工厂函数原型
//凭借于此，暴露在外的工厂函数可以为原型添加方法来为构造函数原型添加方法，所以的实例都可以使用
init.prototype = jQuery.fn;
//暴露工厂函数
w.jQuery = w.$ = jQuery;
}(window));

将入口函数和基本框架融合起来

基础版本

//基本框架
(function (w) {
    // 工厂函数
    function jQuery(selector) {
        return new jQuery.fn.init(selector);
    }
    // 简写原型 + 原型替换
    jQuery.fn = jQuery.prototype = {
        constructor: jQuery
    }
    // 真正的构造函数
    var init = jQuery.fn.init = function (selector) {
        // 过滤空类型
        if (!selector) {
            // 返回空实例
            // 不用设置length
            return this;
        } 
        // 字符串 ==> html || selector
        else if (init.isString(selector)) {
            // trim:优化用户体验，将前后多余空格去除
            selector = init.trim(selector);
            // 判断是否是html
            if (init.isHTML(selector)) {
                // 创建一个临时div
                // 将selector设置为div.innerHTML,这样就自动变为div的子元素
                // 只要遍历子元素分别添加就好
                // 借用数组的push和apply可以简化添加并且可以自动维护length属性
                var tempDiv = document.createElement("div");
                tempDiv.innerHTML = selector;
                [].push.apply(this, tempDiv.childNodes);
            } else {
                // 当做选择器看
                // 会出现选择器无效报错的现象所以try/catch
                try {
                    [].push.apply(this, document.querySelectorAll(selector));
                    // 如果是选择器参数，会有选择器属性值为这个选择器
                    this.selector = selector;
                } catch (e) {
                    // 选择器无效，就这是length为0 没有捕获任何一个元素
                    this.length = 0;
                    this.selector = selector;
                }
            }
        }
        // 数组和伪数组
        else if (init.isLikeArray(selector)) {
            // 如果是数组或者伪数组，每一项分别添加入实例中
            [].push.apply(this, selector);
        } else {
            // 直接添加到实例中，length设置伪1;
            this[0] = selector;
            this.length = 1;

    };

    // 构造函数的静态工具方法
    // 判断是否是html片段
    init.isHTML = funciton (html) {
        // 过滤空类型
        if (!html) {
            return false;
        }
        if (html.charAt(0) == "<" && html.charAt(html.length - 1) === ">" && html.length >= 3) {
            return true;
        }
        return false;
    }

    // 简写
    init._isHTML = function (html) {
        // 利用&&的特性，只要遇到false就立刻返回false，全部为true才为true
        // !!html 双重否定
        return !!html &&
                html.charAt(0) == "<" && 
                html.charAt(html.length - 1) === ">" && 
                html.length >= 3;
    }
    init.trim = function (str) {
        // 不是字符串不作处理 ||过滤null/undefined/NaN/0/false/""
        if (typeof str !== "string" || !str) {
            return str;
        }
        if (str.trim) {
            return str.trim();
        }
        return str.replace(/^\s+|\s+$/g, "");
    }

    init.isWindow = function (w) {
        if (w.window == w) {
            return true;
        }
        return false;
    }
    // 简写
    init._isWindow = function (w) {
        return w.window === w;
    }

    init.isFunction = function (fn) {
        if (typeof fn == "function") {
            return true;
        }
        return false;
    }
    // 简写
    init._isFunction = function (fn) {
        return typeof fn === "function";
    }

    // 判断是否是对象
    init.isObject = function (obj) {
        // 过滤掉null对type的影响
        /*if (obj == null) {
            return false;
        }*/
        // null/undefined/NaN/0/false/""
        if (!obj) {
            return false;
        }
        // typeof对应用类型的由两种object/function，这两种都是对象（null特殊：null存储的是一个空的对象指针）
        if (typeof obj === "function" || typeof obj === "object") {
            return true;
        }
        return false;
    }
    // 判断是否是数组或者伪数组
    init.isLikeArray = funciton (arr) {
        // 先把window/函数和非对象类型数据过滤掉
        if (init.isWindow(arr) || init.isFunction(arr) || !init.isObject(arr)) {
            // window的length属性是指iframe标签的个数
            // 函数length属性是形参个数
            //typeof 数组/伪数组 是object
            return false;
        }
        // 真数组
        if (({}).toString.call(arr) === "[object Array]") {
            return true;
        }
        // 伪书组
        if ("length" in arr && ((arr.length == 0) || (arr.length - 1 in arr))) {
            return true;
        }
        // 其他都不是
        return false;
    }

    // 判断是否是字符床
    init.isString = function (str) {
        if (typeof str === "string") {
            return true;
        }
        return false;
    }
    // 简写
    init._isString = function (str) {
        return typeof str === "string";
    }

    // 替换构造函数原型为工厂函数原型
    // 实现添加插件机制，通过暴露在外的工厂函数原型来为构造函数原型添加成员
    init.prototype = jQuery.fn;
    // 暴露工厂函数和简写
    w.$ = w.jQuery = jQuery;
}(window));

添加了extend方法版本

通过extend方法可以为jQuery和jQuery.fn添加成员

jQuery.extend = jQuery.fn = function (obj) {
    for (var key in obj) {
        if (obj.hasOwnProperty(key)) {
            this[key] = obj[key];
        }
    }
}
