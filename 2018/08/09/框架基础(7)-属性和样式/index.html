attr方法的实现

jQuery.fn.extend( {
    // attr: 获取/设置[属性节点]值
    // 思路一： 先判断参数类型再判断参数类型
    // 过滤掉非字符串/非对象类型的数据类型，直接return this;
    // 如果是字符串类型，判断参数个数
    // 
    attr: function ( attr, val ) {
        // 如果既不是字符串也不是对象数据类型的的就直接returnthis
        if ( !jQuery.isString( attr ) && !jQuery.isObject( attr ) ) {
            return this;
        }
        // 判断类型
        if ( jQuery.isString( attr ) ) {
            // 判断参数个数 一个参数是获取第一个元素的对应属性节点值
            if ( arguments.length === 1 ) {
                // 返回第一个元素的属性节点值
                // dom.getAttribute( attrName )是原生dom方法，获取属性节点值
                return this[ 0 ].getAttribute( attr );
            } else if ( arguments.length >= 2 ) {
                // 遍历设置
                for ( var i = 0, len = this.length; i < len; i++ ) {
                    this[ i ].setAttribute( attr, val );
                }
            }

        } else {
            // 对象类型
            // 外层遍历属性节点值对象，内层遍历实例对象
            for (var key in attr) {
                for ( var i = 0, len = this.length; i < len; i++ ) {
                    this[ i ].setAttribute( key, attr[ key ] );
                }
            }
        }
        // 为了链式编程
        return this;
    },
    // 简写思路二：先判断参数个数，再判断参数类型
    _attr: function ( attr, val ) {
        // 保存一下当前this
        var self = this;
        if ( arguments.length === 1 ) {
            // 判断类型
            if ( jQuery.isString( attr ) ) {
                return this[ 0 ].getAttribute( attr );
            } else if ( jQuery.isObject( attr ) ) {
                // 对象类型需要遍历
                // 使用静态遍历方法
                jQuery.each( attr, function (key, val) {
                    self.each( function () {
                        // 这里的this是实例的每一个元素
                        this.setAttribute( key, val );
                    } );
                } );
            }
        } else if ( arguments.length >= 2 ) {
            this.each( function () {
                this.setAttribute( attr, val );
            } );
        }

        // 链式编程
        return this;
    }
} )

prop方法的实现

jQuery.fn.extend( {
    // prop: 获取/设置[属性]值
    // 思路一： 先判断参数类型再判断参数类型
    // 过滤掉非字符串/非对象类型的数据类型，直接return this;
    // 如果是字符串类型，判断参数个数
    // 
    prop: function ( prop, val ) {
        // 先过滤掉非字符串、非对象的数据类型
        if ( !jQuery.isObject( prop ) && !jQuery.isString( prop ) ) {
            return this;
        }
        // 判断prop的类型，是对象还是字符串，再判断参数个数
        if ( jQuery.isString( prop ) ) {
            if ( arguments.length === 1 ) {
                // 获取第一个元素的对应属性值。通过[]语法
                //这里是获取属性值，所以使用对象的属性获取语法，又因为是变量，所以用[]语法
                return this[ 0 ][ prop ];
            } else if ( arguments.length >= 2 ) {
                // 给所有元素设置
                for ( var i = 0, len = this.length; i < len; i++ ) {
                    this[ i ][ prop ] = val;
                }
            }
        } else {
            // 如果是对象类型 批量设置 双层循环
            for ( var key in prop ) {
                for ( var i = 0, len = this.length; i < len; i++ ) {
                    this[ i ][ key ] = prop[ key ];
                }
            }


        }

        // 链式编程
        return this;
    },
    _prop: function ( prop, val ) {
        var self = this;
        // 先判断个数在类型，并用each来循环
        if ( arguments.length === 1 ) {
            if ( jQuery.isString( prop ) ) {
                // 获取操作
                return this[ 0 ][ prop ];
            } else if ( jQuery.isObject( prop ) ) {
                // 批量设置
                jQuery.each( prop, function ( key, val )  {
                    self.each( function () {
                        this[ key ] = val;
                    } );
                } );
            }
        } else if ( arguments.length >= 2 ) {
            // 设置
            this.each( function () {
                this[ prop ] = val;
            } ); 
        }
        // 链式编程
        return this;
    }
} );

css方法实现

css方法使用：

  1. 当有一个字符串参数的时候获取第一个元素的对应样式值 $(“span”).css( “height” );
  2. 当有一个对象参数的时候，给所有元素设置对象中的所有样式 $(“span”).css( {height: “300px”, width: “100px”} );
  3. 当有两个字符串参数的时候，给所有元素设置对应样式对应值 $(“span”).css( “height” , “200px”);
//为jQuery添加一个获取对应样式值得静态方法
jQuery.extend( {
    getStyle: function ( dom, style ) {
        // 能力测试 如果有getCOmputedStyle就用这个来获取样式
        if ( window.getComputedStyle ) {
            return window.getComputedStyle( dom )[ style ];
        } else {

            // 兼容IE678
            return dom.currentStyle[ style ];
        }
    }
} );

jQuery.fn.extend( {
    // css:判断个数再类型
    css: function ( styleName, style ) {
        if ( arguments.length === 1 ) {
            // 判断是字符串还是对象
            if ( jQuery.isString( styleName ) ) {
                // 返回第一个元素的对应样式值
                return jQuery.getStyle( this[ 0 ], styleName );
            } else if ( jQuery.isObject( styleName ) ) {
                // 双层遍历批量设置
                for ( var key in styleName ) {
                    for ( var i = 0, len = this.length; i < len; i++ ) {
                        //设置写法
                        this[ i ][ "style" ][ key ] = styleName[ key ];
                    }
                }
            }
        } else if ( arguments.length >= 2 ) {
            // 设置
            for ( var i = 0, len = this.length; i < len; i++ ) {
                //设置写法 dom.style[ stylename ]
                //dom[ "style" ][ styleName ];
                this[ i ][ "style" ][ styleName ] = style;
            }
        }
        // 联式编程
        return this;
    },
    //改进写法
    _css: function ( styleName, style ) {
        // 保存一下this值。这里的的是实例
        var self = this;
        if ( jQuery.isString( styleName ) && jQuery.isObject( styleName ) ) {
            return this;
        } 
        // 判断类型再参数个数
        if ( jQuery.isString( styleName ) ) {
            if ( arguments.length === 1 ) {
                // 获取操作
                return jQuery.getStyle( this[ 0 ], styleName );
            } else if ( arguments.length >= 2 ) {
                // 设置操作
                this.each( function () {
                    // 这里的this是每一个元素
                    this[ "style" ][ styleName ] = style;
                } );
            }
        } else {
            // 对象数据类型
            // 双层遍历
            jQuery.each( styleName, function ( key, val ) {

                self.each( function () {
                    // 这里的this是每一个元素
                    this[ "style" ][ key ] = val;
                } );
            } );
        }
        // 链式编程
        return this;
    }
} )

val方法实现

jQuery.fn.extend( {
    // val:不传参数获取第一个元素的value值，传参，设置所有元素的val属性值
    // 链式编程 return this;
    val: function ( value ) {
        if ( arguments.length === 0 ) {
            return this[ 0 ].value;
        } else {
            for ( var i = 0, len = this.length; i < len; i++ ) {

                this[ i ].value = value;
            }
        }
        // 链式编程
        return this;
    },
    // each简写
    _val: function ( value ) {
        if ( arguments.length === 0 ) {
            return this[ 0 ].value;
        } else {
            this.each( function () {
                this.value = value;
            } );
        }
        return this;
    },
    // 借用prop
    __val: function ( value ) {
        if ( arguments.length === 0 ) {
            // 没有传参，借用prop返回第一个元素的value属性值
            return this.prop( "value" );
        } else {
            // 传参数。给所有元素的value属性值设置，并返回当前实例
            return this.prop( "value", value )
        }
    }
} );

hasClass(四种写法)

jQuery.fn.extend( {
    // hasClass:判断实例中元素是否含有某个类，如果其中一个有就返回true,全都没有才返回false
    hasClass: function ( classname ) {
        for ( var i = 0, len = this.length; i < len; i++ ) {
            // 处理每一个元素的className
            var domClass = " " + this[ i ].className + " ";
            // 对传入的参数也做前后加空格处理
            // 如果indexOf的返回值不等于 -1 说明存在该类，返回true
            if ( domClass.indexOf( " " + classname + " " ) > -1 ) {
                return true;
            }
        }
        // 整个循环完毕都没有返回true ，也就是说所有元素都没有这个类
        return false;
    },
    // each改写
    _hasClass: function ( classname ) {
        // 用来标记是由含有该类，默认没有
        var has = false;
        this.each( function () {
            // ！== -1不严谨，因为undefined、null
            if ( ( " " + this.className + " " ).indexOf( " " + classname + " " ) > -1 ) {
                // 为什么这里不是return true
                // 因为这里中断的是each中的回调，而不是hasClass这个方法，也就是说
                // 无法通过这里的return true获得想要的判断值
                // 最后 无论如何都会执行 代码块最下方的return false
                // return true;

                // 使用标记法
                has = true;
            }
        } );
        // return false;
        return has;
    },
    // 改进第二个方法，当获得true后中断遍历
    __hasClass: function ( classname ) {
        // 默认是没有
        var has = false;
        this.each( function () {
            if ( ( " " + this.className + " " ).indexOf( " " + classname + " " ) > -1  ) {
                has = true;
                // 中断遍历 each方法底层封装了，当获得false时，会break
                return false;
            }
        } );
        return has;
    },
    // 测试是否可以用reg和match
    ___hasClass: function ( classname ) {
        var has = false;
        // classname去空格处理
        classname = jQuery.trim( classname );
        var classReg = new RegExp( "\\b" + classname + "\\b" );
        this.each( function () {
            // 这里的this是每一个元素
            //str.match:查找字符串中是否有匹配正则参数的字符串，有的话返回一个数组存放匹配的字符串
            //没有的话返回null
            if ( this.className.match( classReg ) !== null ) {
                has = true;
                // 只要有一个有就终止遍历
                return false;
            }
        } );
        return has;
    }
} )；

addClass

jQuery.fn.extend( {
    // addClass：遍历实例中每一个元素，
    //             判断这个元素是否有这个class
    //             如果有就忽略
    //             没有就给它添加上这个class
    addClass: function ( classname ) {
        // this ==> jQObj
        this.each( function () {
            // this ==> dom in jQobj

            // does it has this classname
            if ( !jQuery( this ).__hasClass( classname ) ) {
                // if not: add 
                this.className += " " + classname;
            }
        } );
        // 链式编程
        return this;
    },
    // 批量处理利用str.split将字符串参数转为数组
    // 双层遍历：外层遍历实例中的每一个dom元素
    //             内层遍历每一个class
    _addClass: function ( classnames ) {
        // 对传入的参数进行处理，去掉可能存在的前后空格，在以空格为依据劈成数组
        var classArr = jQuery.trim( classnames ).split( " " );
        // 双层遍历
        // 外层遍历每一个元素
        this.each( function () {
            // this ==> dom
            // 之后需要使用jq实例方法，所以进行jq包装
            var $this = jQuery( this );

            // 内层遍历classArr
            jQuery.each( classArr, function ( i, val ) {
                // 判断当前元素是否有这个class,有就忽略，无则添加
                if ( !$this.__hasClass( val ) ) {
                    // 这里通过dom.className来添加新的class
                    // 所以要将jq实例转为dom
                    $this[0].className += " " + val;
                }
            } );
        } );

        // 链式编程
        return this;
    }
} )；

removeClass

这个方法的改进使用了动态创建正则表达式

jQuery.fn.extend( {
    // removeClass:如果没有传参数
    //                 遍历每一个元素清空className
    //                 如果传了阐述 便利每一个元素清除对应class(使用前后空格处理法和replace)
    //                 链式编程 返回this
    removeClass: function ( classname ) {
        if ( arguments.length === 0 ) {
            this.each( function () {
                this.className = "";
            } );
        } else {
            this.each( function () {
                this.className = ( " " + this.className + " " ).replace( " " + classname + " ", " " );
                // 借用静态方法去一下前后空格
                this.className = jQuery.trim( this.className );
            } );
        }
        return this;
    },
    // removeClass: 升级split + 正则
    _removeClass: function ( classnames ) {
        // 先判断参数个数
        if ( arguments.length === 0 ) {
            // 将每个元素的className清空
            this.each( function () {
                this.className = "";
            } );
        } else {
            // 有参数则清除每一个元素中每一个需要清除的class
            // 处理参数，将其变为一个可以遍历的数组
            var classArr = jQuery.trim( classnames ).split( " " );
            // 双层遍历，外层遍历每一个元素，内层遍历classArr
            this.each( function () {
                // 这里的this是每一个元素
                var self = this;

                jQuery.each( classArr, function ( i, val ) {
                    // !!!这里改进是使用正则配合replace而不是空格处理
                    self.className = self.className.replace( new RegExp( "\\b" + val + "\\b" ), "" );
                } );

            } );
        }
        // 链式编程
        return this;
    }
} )；

toggleClass

jQuery.fn.extend( {
    // toggleClass:遍历所有元素，如果有这个class就删除，没有就添加 链式编程返回this
    toggleClass: function ( classname ) {
        this.each( function () {
            // 这里的this是实例中的每一个dom元素
            //为了使用jq实例方法，而包装一下
            var $this = jQuery( this );
            if ( $this.__hasClass( classname ) ) {
                $this.removeClass( classname );
            } else {
                $this.addClass( classname );
            }
        } );
        return this;
    },
    // 这里的改进也是劈成数组双层遍历
    _toggleClass: function ( classnames ) {
        var classArr = jQuery.trim( classnames ).split( " " );
        this.each( function () {
            // 这里的this是dom，之后需要使用jq实力方法，所以包装
            var $this = jQuery( this );
            jQuery.each( classArr, function ( i, val ) {
                // 判断当前元素是否有这个class，有则删除，没有添加
                if ( $this.__hasClass( val ) ) {
                    $this._removeClass( val );
                } else {
                    $this._addClass( val );
                }
            } );
        } );
        // 链式编程
        return this;
    }
} )；
