jQuery原型核心属性和方法

属性

  1. jQuery：版本号
  2. selector：所有实例的默认选择器
  3. length：所有实例默认的长度

方法

  1. toArray: 将实例转变为数组
  2. get: 按照下标获取实例的元素，返回的是一个原生DOM元素
  3. eq: 同上，但返回的是一个新的实例对象
  4. each：遍历实例，将实例的下标和元素传入回调函数中
  5. map: 同上，并且会将回调的返回值组成一个新数组，并返回
  6. slice: 从实例中截取一部分组成一个新实例并返回
  7. first: 获取实例第一个元素，并返回的是一个新实例对象
  8. last: 获取实例最后一个元素，返回一个新实例对象
  9. push: 为实例添加一个元素
  10. sort: 在实例中按照某个规则将实例元素排序
  11. splice: 按照指定下标指定数量删除/替换元素

jQuery原型中上述属性和方法的实现

实例成员部分

jQuery.fn = jQuery.prototype = {
    constructor: jQuery,
    // 版本
    jQuery: version,
    // 所有实例默认选择器，并且是jQuery实例的标志
    selector: "",
    // 所有实例默认长度
    length: 0,

    // 核心方法
    // 将jq实例转为一个数组，并放回
    toArray: function () {
        return [].slice.call( this );
    },
    // 获取指定下标的实例元素，获取的是DOM原生元素
    get: function ( i ) {
        if ( i == null ) {
            // == 情况下undefined与null 一样
            // null、undefined返回一个实例转成的数组
            return this.toArray();
        }
        // 正负数
        if ( i >= 0 ) {
            return this[i];
        } else {
            return this[ this.length + i ];
        }
    },
    // 简写
    _get: function ( i ) {
        return i == null ?
                this.toArray() :
                ( i >= 0 ? this[ i ] : this[ this.length + i ]);
    },
    // 下面两个方法通过之后编写的静态方法来实现
    each: function ( fn ) {
        return jQuery.each( this, fn );
    },
    map: function ( fn ) {
        return jQuery.map( this, fn );
    },
    // 截取部分实例对象元素，组成一个新实例对象并返回
    slice: function () {
        // 数组.slice返回的是一个截取出来的新数组
        // 需要通过工厂函数包装才能成为一个jQuery实例对象
        // 由于slice方法接受的参数不定， 所以这里通过argument来接收参数
        var nodes = [].slice.apply( this, arguments );
        return jQuery( nodes );
    },
    // 简写
    _slice: function () {
        return jQuery( [].slice.apply( this, arguments ) );
    },
    // 获取指定下标的元素，获取的是实例对象
    eq: function ( i ) {
        if ( i == null ) {
            // null/undefined返回一个新的空实例
            return jQuery();
        }
        // num
        if ( i >= 0 ) {
            return jQuery( this[ i ] );
        } else {
            return jQuery( this[ this.length + i ] );
        }
    },
    // 简写
    _eq: function ( i ) {
        return i == null ? jQuery() : jQuery( this.get( i ) );
    }
    // 获取实例对象的第一个元素，并返回一个新的实例
    first: function () {
        return this.eq( 0 );
    },
    // 获取实例对象的最后一个元素，并返回一个新的实例
    last: function () {
        return this.eq( -1 );
    },
    //原型方法由实例调用，当调用时this指向就是实例自身
    //所以不需要call/apply来改变this指向
    push: [].push,
    sort: [].sort,
    splice: [].splice

}

    each: function ( obj, fn ) {
        var i = 0, len, key;
        // 数组、伪数组使用i
        if ( jQuery.isLikeArray( obj ) ) {
            len = obj.lengthl
            for( ; i < len; i++ ) {
                // 遍历终止 + this指向改变
                //如果不改变this指向的话，this指向的是window,这里需要this指向遍历的val
                // each中回调参数 index val
                // 其他数组遍历参数都是 value index
                if ( fn.call( obj[ i ], i, obj[ i ] ) === false ) {
                    break;
                }
            }
        } else {
            // 对象 for in
            for ( key in obj ) {
                // 遍历终止 + this指向改变
                // each中回调参数 index val
                // 其他数组遍历参数都是 value index
                if ( fn.call( obj[ key ], key, obj[ key ] ) === false ) {
                    break;
                }
            }
        }

        // 为了之后的链式编程，会返回obj
        return  obj;
    },

    // map:遍历并将value和index传入回调，并将回调返回值组成数组返回
    map: function ( obj, fn ) {
        var i = 0, len, key, result, results = [];
        if ( jQuery.isLikeArray( obj ) ) {
            len = obj.length;
            for ( ; i < len; i++ ) {
                result = fn.call( obj[ i ], obj[ i ], i );
                if ( result != null ) {
                    results.push( result );
                }
            }
        } else {
            for ( key in obj ) {
                result = fn.call( obj[ key ], obj[ key ], key );
                if ( result != null ) {
                    results.push( result );
                }
            }
        }

        // 返回结果数组
        return results;
