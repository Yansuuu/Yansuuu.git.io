冒泡排序：

排序的一种，相邻两个元素作比较，符合某种规则交换位置，直到达到目标为止

一下为利用冒泡排序将一组数列从小到大重新排序

版本一：原始版对比量为（n-1）* （n-1）

var arr = [1, 22, -5, 6, 44, 8]; 
for (var i = 0; i < arr.length-1; i++) {
//第一版冒泡排序将对比n-1轮，每轮对比n-1次，所以用一个循环控制轮数;
//第二个循环控制每轮次数;
//n为数组元素个数，即arr.length;
//因为数组索引取值从0开始，所以arr.length-1;
    for (var j = 0; j < arr.length-1; j++) {
        if (arr[j] > arr[j+1]) {
            //把大的数放到后面去，小的放到前面去
            var temp = arr[j+1];
            arr[j+1] = arr[j];
            arr[j]  = temp;
        }
    }
}
console.log(arr) //[-5, 1, 6, 8, 22, 44]

版本二：减少每轮比较次数

实际上在对比时，每一轮都会得到自己一个最值（最大值或者最小值），

也就是说每轮都可以比上一轮少比较一次

var arr = [1, 22, -5, 6, 44, 8]; 
for (var i = 0; i < arr.length-1; i++) {
//第二版冒泡排序还是对比n-1轮
    //但每轮对比次数都比上一轮少一次,所以j < arr.lenegth-1-i;
    for (var j = 0; j < arr.length-1-i; j++) {
        if (arr[j] > arr[j+1]) {
            //把大的数放到后面去，小的放到前面去
            var temp = arr[j+1];
            arr[j+1] = arr[j];
            arr[j]  = temp;
        }
    }
}
console.log(arr) //[-5, 1, 6, 8, 22, 44]

第一版对比轮数是5*5=25次；

第二版则是5+4+3+2+1=15次；

版本三：在减少每轮对比次数的基础上，减少轮数

实际将数组重新排序的情况，会有在中途就已经排序好的情况，如果我们能检查出来这种情况，剩下的比较轮数就可以不用去进行了。
什么情况算是排序好了呢？
当一轮比较中不在有前边的数比后面的数大了，不再发生换位的时候。

var arr = [1, 22, -5, 6, 44, 8]; 
var sum1 = 0; //来统计一共比较了多少次
var sum2 = 0; //来统计一共比较了几轮
for (var i = 0; i < arr.length-1; i++) {
//第三版冒泡排序将使用开闭原则检查是否还需要换位
    var flag = true //设置一个标识来判断是否发生了换位
    for (var j = 0; j < arr.length-1-i; j++) {
        if (arr[j] > arr[j+1]) {
            var temp = arr[j+1];
            arr[j+1] = arr[j];
            arr[j]  = temp;
            flag = false; //发生了换位将标识flag赋值为false；
        }
        sum1 =+ 1;
    }
    sum2 += 1;
    if (flag) {
        break; //当flag不发生变化，也就是说这一轮中的元素已经不发生交换了，这时候可以跳出循环，因为排序已经完成了；
    }
}
console.log(arr);
//[-5, 1, 6, 8, 22, 44];
console.log(sum1); //12
console.log(sum2); //3

第三版只比较了三轮，12次，比第二版比较次数还要少。
