缓动动画框架简易版本封装二

Yansuuu 5月 08, 2018
  • javascript
  • 练习

这次的缓动动画封装升级部分如下：

  1. 传入一个json，可在动画中变化多个属性
  2. 针对透明度和层级做了优化
  3. 针对清除定时器做了优化
  4. 可传递回调函数

    <script type="text/javascript">
        //老三步
        var btn = document.getElementsByTagName("button")[0];
        var box = document.getElementById("box");
        //给 按钮绑上点击事件
        btn.onclick = function ()  {
            var json1 = {
                "width": 300, "height": 300,
                "top": 200, "left": 250,
                "border-width": 20, "border-radius": 170,
                "opacity": 90, "zIndex": 1};
            var json2 = {
                "width": 100, "height": 100,
                "top": 40, "left": 0,
                "border-width": 3, "border-radius": 53,
                "opacity": 100, "zIndex": 0};

            animate(box, json1, function () {
                animate(box, json2)
            })
        }

        //封装一个缓动动画框架，参数分别是要动的盒子，传入的样式json,和回调函数
        function animate(ele, json, fn) {
            //用前清理定时器
            clearInterval(ele.timer);
            ele.timer = setInterval(function () {
                //通过开闭原则来为后续清理定时器做准备
                var flag = true;
                for (var k in json) {
                    var leader;
                    //如果属性是透明度的话做特殊处理
                    if (k === "opacity") {
                        //由于透明的的值是小数，计算会有误差，所以将其*100
                        leader = getStyle(ele, k) * 100 || 100;
                    } else {
                        leader = parseInt(getStyle(ele, k)) || 0;
                    }
                    //步长和二次处理
                    var step = (json[k] - leader) / 10;
                    step = step > 0 ? Math.ceil(step) : Math.floor(step);

                    //赋值（注意透明度和层级）
                    if (k === "zIndex") {
                        //层级不作缓动处理，直接赋值
                        ele.style.zIndex = json[k];
                    } else if (k === "opacity") {
                        ele.style[k] = (leader + step) / 100;
                    } else {
                        ele.style[k] = leader + step + "px";
                    }

                    //清理定时器第一步
                    if (Math.abs(json[k] - leader) > Math.abs(step)) {
                        //在这里判断是否还有属性需要继续向目标值移动，如果有，
                        //就将标识符改为false，阻止之后的清理定时器
                        flag = false;
                    }
                }

                //清理定时器第二步
                //这里可以在控制台看是否清除了定时器
                console.log(1);
                if (flag) {
                    clearInterval(ele.timer);
                    //这里是为了防止一些含有小数的目标值走不到位
                    for (var k in json) {
                        if (k === "opacity") {
                            ele.style[k] = json[k] / 100;
                        } else {
                            ele.style[k] = json[k] + "px";
                        }
                    }

                    //回调函数部分
                    //如果有回调函数就在上一个函数所有变化都结束后进行
                    if (fn) {
                        fn();
                    }
                }

            }, 20)
        }

        //先封装一个获取属性函数
        function getStyle(ele, attr) {
            if (window.getComputedStyle) {
                return window.getComputedStyle(ele, null)[attr];
            } else {
                //兼容IE678
                return ele.currentStyle[attr];
            }
        }
    </script>
