AJAX 2.0: FormData 与 Upload 与 FileReader

案例介绍：实现将文件拖拽到方块中异步上传的功能

要点：

  • 使用 formdata 来模拟表单
  • 配合 XMLHttpRequest 异步上传
  • 使用拖拽 api (e.dataTransfer.files)
  • 使用 xhr.upload.onprogress 来实现进度条功能

HTML 页面

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <meta http-equiv="X-UA-Compatible" content="ie=edge">
  <style>
    .box {
      width: 200px;
      height: 150px;
      background: #666;
      color: #fff;
      text-align: center;
      line-height: 150px;
      position: absolute;
      left: 0;
      right: 0;
      top: 0;
      bottom: 0;
      margin: auto;
    }
    .parent {
      width: 60%;
      border: 1px solid #333;
      height: 30px;
    }
    .son {
      background: #64ffda;
      height: 100%;
      width: 0;
    }
  </style>
  <title>ajax.upload</title>
</head>
<body>
  <div class="parent">
    <div class="son"></div>
  </div>
  <div class="box" id="box">一个单纯的盒子</div>

  <script>
    const box = document.querySelector('#box');
    const son = document.querySelector('.son');
    // 实现拖拽上传文件 配合 拖拽 api 和 formdata
    document.addEventListener('dragenter', function(e) {
      box.innerHTML = '拖拽到这里';
    });
    box.addEventListener('dragleave', function(e) {
      box.innerHTML = '一个单纯的盒子';
    });

    box.addEventListener('dragover', function(e) {
      box.innerHTML = '在这里放下上传';

      // 必须阻止浏览器默认事件 否则无法触发 drop 事件
      // 在 onxx = function 中使用 return false 就可
      // 在这种 回调函数中 使用 e.preventDefault
      e.preventDefault();
    });

    box.addEventListener('drop', function(e) {
      // 这里也要阻止浏览器默认事件 防止浏览器自动打开拖拽过来的文件

      // 在这里使用 formdata 和 ajax
      const data = new FormData();
      // 拖拽的文件的数据存储在 e.dataTransfer.files 中 不是个数组
      // console.log(e.dataTransfer.files instanceof Array);

      Array.from(e.dataTransfer.files).forEach(file => {
        data.append('f1', file);
      });

      const xhr = new XMLHttpRequest();

      // post
      xhr.open('post', '/upload', true);

      // xhr.upload.onprogress 必须放在 send 之前 因为要发送两次请求
      // 一次 options 一次 post
      xhr.upload.addEventListener('progress', function (e) {
        // e.loaded 加载
        // e.total 总大小
        son.style.width = e.loaded / e.total * 100 + '%';

      });

      // 使用 formdata 来上传文件不用设置 请求头
      // xhr.setRequestHeader('Content-Type', 'multipart/form-data');
      // 直接发送 添加好的 formdata 实例对象
      xhr.send(data);

      /* xhr.onreadystatechange = function() {
        if (xhr.readyState === 4) {
          if (xhr.status >= 200 && xhr.status < 300 || xhr.status === 304) {
            console.log(xhr.responseText);
          }
        }
      } */

      e.preventDefault();
    });
  </script>
</body>

express 搭建的服务器

// 使用 框架处理文件上传 原生很麻烦啦
const express = require('express');
const formidable = require('express-formidable');

const app = express();

app.listen(3000, () => {
  console.log('this server is running');
});


// 配制跨域 cros
app.all('*', (req, res, next) => {
  if (!req.get('origin')) return next();
  // 设置响应头
  // 关键是 这个 Access-Control-Allow-Origin
  res.set('Access-Control-Allow-Oringin', '*');
  // res.set('Access-Control-Allow-Methods', 'GET')
  if ('OPTIONS' === req.method) return res.send(200);
  // 向下匹配
  next();
});

// 开启静态资源文件夹
app.use(express.static('./www'));


// 使用插件才能使用 rer.files 来获取上传的文件
app.use(formidable({
  encoding: 'utf-8',
  uploadDir: './upload',  // 上传文件存储到的文件夹
  multiples: true // req.files to be arrays of files
}));

app.post('/upload', (req, res) => {

  console.log(req.files);
  res.send('ok');

});

xhr.upload 使用注意

必须在 xhr.send之前

原因：请求发送两次 ，第一次是 options 第二次是 post。如果将 xhr.upload 放在 send 之后，会报错

说你在发送内容之后设置请求头

实现上传图片预览功能

<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <meta http-equiv="X-UA-Compatible" content="ie=edge">
  <style>
    ul,
    li {
      margin: 0;
      padding: 0;
    }

    ul {
      width: 60%;
      /* border: 1px solid #333; */
      list-style: none;
      overflow: hidden;
      position: relative;
      left: 0;
      right: 0;
      top: 0;
      bottom: 0;
      margin: auto;
    }

    .list {
      float: left;
      width: 20%;
      position: relative;
      margin-right: 15px;
    }

    .list img {
      width: 100%;
    }

    .del {
      position: absolute;
      top: 0;
      right: 0;
      text-decoration: none;
      color: #fff;
      font-size: 18px;
      width: 20px;
      height: 20px;
      background: rgba(255, 255, 255, 0.699);
      text-align: center;
      line-height: 20px;
    }

    .box {
      width: 150px;
      height: 120px;
      border: 1px solid #333;
      text-align: center;
      line-height: 120px;
      position: absolute;
      left: 0;
      right: 0;
      top: 0;
      bottom: 0;
      margin: auto;
    }
    .box > span {
      position: absolute;
      bottom: 0px;
      left: 48px;
      line-height: 28px;
      border: 1px solid #333;
      cursor: pointer;
    }
  </style>
  <title>FileReader</title>
</head>

<body>

  <ul id="ulBox">
  </ul>
  <div class="box">
      拖到这里
      <span id='uploadBox'>上传</span>
  </div>

  <script>
    const ulBox = document.querySelector('#ulBox');
    const box = document.querySelector('.box');
    const uploadBox = document.querySelector('#uploadBox');

    box.addEventListener('drapenter', function() {
      this.innerHTML = 'HI';
    });
    box.addEventListener('drapleave', function() {
      this.innerHTML = '拖到这里';
    });

    box.addEventListener('dragover', function(e) {
      // 防止浏览器阻止 drop 事件
      e.preventDefault();
    });

    box.addEventListener('drop', function(e) {
      // 防止浏览器打开文件
      e.preventDefault();
      Array.from(e.dataTransfer.files).forEach(file => {
        // file.type: image/png  text/html ...
        // 不是图片就不读取了
        if (!file.type.startsWith('image/')) {
          console.log('请传入图片');
          return;
        }
        /* 将每一个拖拽的文件-这里是图片 生成一个 li 含有图片和删除按钮
          1. 创建 FileReader 实例对象
          2. reader.onload 后
          3. 创建 li
          4. 将 file 赋值给li.file 作为保存, 之后上唇需要

        */
        // 创建 filereader 实例对象
        const reader = new FileReader();
        // 实例对象有一系列完整的事件 读取成功事件 onload
        reader.onload = function() {
          const li = document.createElement('li');
          // 将当前这个拖拽的文件存一个指针在当前创建的这个 li 上
          li.file = file;
          li.className = 'list';
          li.innerHTML = `<img>
      <a href="javascript:;" class="del">×</a>`

          // li 中的图片
          const img = li.children[0];
          // reader.result 是 filereader 读取的结果存储的地方
          img.src = this.result;

          const delBtn = li.children[1];
          delBtn.onclick = function() {
            // 将这个 Li 移除
            ulBox.removeChild(li);
          }

          // 将 li 添加到页面中
          ulBox.appendChild(li);
        }

        //  读成 base64
        reader.readAsDataURL(file);
      });
    });

    // 上传到服务器
    uploadBox.onclick = function () {
      const formData = new FormData();
      Array.from(ulBox.children).forEach(li => {
        // 在 filereader 的 onload 事件中存储的文件指针
        formData.append('pic', li.file);
      });

      const xhr = new XMLHttpRequest();
      xhr.open('post', '/upload', true);
      xhr.send(formData);

      xhr.onreadystatechange = function () {
        if (xhr.readyState === 4) {
          if (xhr.status >= 200 && xhr.status < 300 || xhr.status === 304) {
            console.log(xhr.responseText);
          }
        }
      }
    }
  </script>
</body>

</html>
