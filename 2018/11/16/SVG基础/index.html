SVG 基础

基础部分(常用)

特点：

  • 矢量图
  • 事件监听
  • 可以修改
  • 性能一般

基础标签：

  • 线

  <line x1='100' y1='100' x2='200' y2='100' stroke='black' stroke-width='5'>
</svg>

  • 圆 cx. cy 圆心坐标 r 半径

<svg width="800" height="600">
    <circle cx='400' cy='300' r='200'></circle>
</svg>
  • 椭圆

  • 矩形

    <rect x='100' y='100' width='300' height='200' style="fill:aqua; stroke:#000; stroke-width:5; opacity: 0.5;" onclick='alert("a")'></rect>

  • 路径

svg 饼图封装

<!-- html部分 -->
<svg id="svg" width='800' height='600'></svg>


// js 部分
const svg = document.getElementById('svg');
// 角度 --》 弧度
function d2a(deg) {
  return deg * Math.PI / 180;
}

// 根据角度和圆心，半径确定在圆上的一点坐标
function getPoint(x, y, r, ang) {
  return {
    x: x + r * Math.sin(d2a(ang)),
    y: y - r * Math.cos(d2a(ang))
  }
}

// 绘制扇形
function drawSector(p, cx, cy, r, ang, oldAng, color) {
  let dArr = [];

  // 1.移动到圆心，画到起始点
  let { x: x1, y: y1 } = getPoint(cx, cy, r, oldAng);
  dArr.push(`M ${cx} ${cy} L ${x1} ${y1}`);

  // 2.画弧
  let { x: x2, y: y2 } = getPoint(cx, cy, r, ang + oldAng);
  dArr.push(`A ${r} ${r} 0 ${ang >= 180 ? 1 : 0} 1 ${x2} ${y2}`);

  // 闭合
  dArr.push('Z');

  // 拼接字符串，设置属性
  p.setAttribute('d', dArr.join(' '));

  // 生成随机填充颜色 0xFFFFFF === 16777215
  // str.padStart(targetLeng, padStr) 在字符串前方根据指定字符串填充到指定位数
  color = color || Math.floor(Math.random() * 16777216).toString(16).padStart(6, 0);
  p.style.fill = color;

  // 将获取到的 color 绑到这个 p 上
  p.color = color;
}
// 绘制饼图：计算数据综合，根据每份数据的占比获得每份 pie 的角度，绘制
function drawPie(dataArr, cx = 400, cy = 300, r = 150) {
  // 计算数据和
  let sum = dataArr.reduce((pre, cur) => {
    return pre + cur;
  });
  // 计算基础角度
  let baseAng = 360 / sum;
  let oldAng = 0
  dataArr.forEach(data => {
    // 创建 path 并添加到 svg 中
    let pie = document.createElementNS('http://www.w3.org/2000/svg', 'path');
    svg.appendChild(pie);

    // 计算当前数据所占角度，并绘制扇形
    let ang = data * baseAng;
    drawSector(pie, cx, cy, r, ang, oldAng);

    // 添加鼠标移动悬放、移出事件
    // 保留当前值
    pie.oldAng = oldAng;
    pie.onmouseover = function() {
      pieMove(this, r * 1.5, r, ang, this.oldAng);
    }
    pie.onmouseleave = function() {
      pieMove(this, r, r * 1.5, ang, this.oldAng);
    };
    oldAng += ang;
  });

}

// 饼图动画封装
function pieMove(p, targetR, r, ang, oldAng, cx = 400, cy = 300) {
  // 从 r --> targetR
  // 做成动画版本
  let disR = targetR - r,
    // 动画帧数
    frames = 40,
    // 帧数计数器
    frameCounter = 0;

  requestAnimationFrame(next);
  // 缓动
  function next() {
    let a = 1 - frameCounter / frames
    let cur = r + disR * (1 - Math.pow(a, 3));
    drawSector(p, cx, cy, cur, ang, oldAng, p.color);
    frameCounter += 1;
    // 当帧数计数器和帧数一致的时候，也就是说规定帧数已经走完，就无需调用 raf 函数了
    // 这个动画也就停止了
    if (frameCounter < frames) {
      requestAnimationFrame(next);
    }
  }

}

// 当前饼图的圆心坐标，半径，和数据
let cx = 400,
  cy = 300,
  r = 150,
  datas = [100, 150, 300, 75, 210, 30];
// 绘制
drawPie(datas)

RaphaelJS

一个兼容 svg 和 vml 的矢量图形库，可以兼容到 ie 低版本浏览器

使用：类似 jquery 可以链式操作

  • attr: setAttribute/getAttribute
  • click: 绑定点击事件
  • …

基础使用

<script src="./raphael.js"></script>
<script>
  // 自动创建一个 svg 在页面中，使用绝对定位
  // x, y, w, h
  const paper = Raphael(0, 0, 800, 600);

  paper.rect(100, 100, 300, 200, 10)
    .attr({
      fill: 'cyan',
      stroke: '#e42',
    })
    .click(function () {
      this.animate({
        width: 400,
        fill: 'hotpink'
      }, 1000, 'bounce')
    })
  // 简单的动画效果 三 ==> X
  const path = paper.path('M 200 100 L 300 100 M 200 120 L 300 120 M 200 140 L 300 140')
      .attr({
        'stroke-width': 10,
        'stroke': '#64FFDA',
        fill: 'none'
      });
  path.click(function() {
    this.animate({
      path: 'M 200 100 L 300 140 M 200 120 L 200 120 M 200 140 L 300 100'
    }, 1000, 'bounce');
  });
</script>

D3

非常厉害厉害厉害的库，功能十分强大，所以一般人用不了，好好学数学后才学吧

简单使用

各种生成器

一个扇形绘制

<script src="./d3.js"></script>
<script>
  function color() {
    return `#${Math.floor(Math.random() * 16777216).toString(16).padStart(6, '0')}`;
  }
  // console.log(color());

  let datas = [
    { name: '1月', value: 200, color: color() },
    { name: '2月', value: 125, color: color() },
    { name: '3月', value: 77, color: color() },
    { name: '4月', value: 142, color: color() },
  ]
  const g = d3.select('body').append('svg')
    .attr('width', '800px')
    .attr('height', '600px')
    // g => group 在 svg 规范里用来分组
    .append('g');

  // pie 数据生成器 根据数组生成扇形数据数组
  const pieGen = d3.pie()
    .value(json => json.value)
    // 和数组排序方式一样
    .sort((j1, j2) => 0)
    .padAngle(() => {
      // 扇形之间是否要有空隙
      return 2 * Math.PI / 180
    })

  // arc 绘画 d 属性值生成器
  const arcGen = d3.arc()
    // 内外半径
    .innerRadius(60)
    .outerRadius(200)
    // 圆角
    .cornerRadius(20)

  const pieRes = pieGen(datas);
  // 根据这个数据循环绘画
  pieRes.forEach(pie => {
    // console.log(pie);

    // 生成这个扇形的 d 属性值
    let arcRes = arcGen(pie);
    g.append('path')
      .attr('d', arcRes)
      .attr('fill', pie.data.color);
  });

  // 整体移动 g
  g.attr('transform', 'translate(400, 300)')
</script>
